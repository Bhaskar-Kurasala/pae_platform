```python
import uuid
from sqlalchemy import String, Text, DateTime
from sqlalchemy.orm import Mapped, mapped_column
from app.models.base import Base
from datetime import datetime, UTC

class Feedback(Base):
    __tablename__ = "feedback"

    id: Mapped[uuid.UUID] = mapped_column(primary_key=True, default=uuid.uuid4)
    user_id: Mapped[uuid.UUID | None] = mapped_column(nullable=True, index=True)
    route: Mapped[str] = mapped_column(String(255))
    body: Mapped[str] = mapped_column(Text)
    sentiment: Mapped[str | None] = mapped_column(String(20), nullable=True)  # "positive"|"neutral"|"negative"
    resolved: Mapped[bool] = mapped_column(default=False)
    created_at: Mapped[datetime] = mapped_column(
        DateTime(timezone=True), default=lambda: datetime.now(UTC)
    )
```

- [ ] **Step 9: Create feedback route**

```python
# backend/app/api/v1/routes/feedback.py
import uuid
import structlog
from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from app.core.database import get_db
from app.core.security import get_current_user
from app.models.feedback import Feedback
from app.models.user import User
from app.schemas.feedback import FeedbackCreate, FeedbackItem

log = structlog.get_logger()
router = APIRouter(prefix="/feedback", tags=["feedback"])

def _require_admin(current_user: User = Depends(get_current_user)) -> User:
    if current_user.role != "admin":
        raise HTTPException(status_code=status.HTTP_403_FORBIDDEN, detail="Admin only")
    return current_user

@router.post("", status_code=201)
async def submit_feedback(
    body: FeedbackCreate,
    db: AsyncSession = Depends(get_db),
    current_user: User | None = Depends(get_current_user),  # optional auth
) -> dict:
    feedback = Feedback(
        user_id=current_user.id if current_user else None,
        route=body.route,
        body=body.body,
        sentiment=body.sentiment,
    )
    db.add(feedback)
    await db.commit()
    log.info("feedback.submitted", route=body.route, sentiment=body.sentiment)
    return {"id": str(feedback.id)}

@router.get("/admin", response_model=list[FeedbackItem])
async def list_feedback(
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> list[FeedbackItem]:
    result = await db.execute(
        select(Feedback).order_by(Feedback.created_at.desc()).limit(200)
    )
    items = result.scalars().all()
    log.info("admin.feedback_listed", count=len(items))
    return [FeedbackItem.model_validate(item, from_attributes=True) for item in items]

@router.patch("/admin/{feedback_id}/resolve", status_code=204)
async def resolve_feedback(
    feedback_id: uuid.UUID,
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> None:
    result = await db.execute(select(Feedback).where(Feedback.id == feedback_id))
    item = result.scalar_one_or_none()
    if not item:
        raise HTTPException(status_code=404, detail="Feedback not found")
    item.resolved = True
    await db.commit()
    log.info("admin.feedback_resolved", feedback_id=str(feedback_id))
```

```tsx
// frontend/src/components/features/feedback-widget.tsx
"use client";

import { useState } from "react";
import { MessageSquare, X, Send } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { usePathname } from "next/navigation";

export function FeedbackWidget() {
  const [open, setOpen] = useState(false);
  const [body, setBody] = useState("");
  const [sent, setSent] = useState(false);
  const pathname = usePathname();

  const submit = async () => {
    if (!body.trim()) return;
    await fetch("/api/v1/feedback", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ route: pathname, body }),
    });
    setSent(true);
    setTimeout(() => { setOpen(false); setSent(false); setBody(""); }, 2000);
  };

  return (
    <div className="fixed bottom-4 right-4 z-50">
      {open ? (
        <div className="w-72 rounded-lg border border-border bg-card p-4 shadow-lg">
          <div className="flex items-center justify-between mb-2">
            <span className="text-sm font-medium">Send feedback</span>
            <button onClick={() => setOpen(false)} aria-label="Close feedback">
              <X className="h-4 w-4 text-muted-foreground" />
            </button>
          </div>
          {sent ? (
            <p className="text-sm text-green-600">Thanks! Feedback sent.</p>
          ) : (
            <>
              <Textarea
                placeholder="What's on your mind?"
                value={body}
                onChange={(e) => setBody(e.target.value)}
                rows={3}
                className="mb-2 text-sm"
                aria-label="Feedback message"
              />
              <Button size="sm" onClick={submit} disabled={!body.trim()} className="w-full">
                <Send className="mr-1.5 h-3.5 w-3.5" />
                Send
              </Button>
            </>
          )}
        </div>
      ) : (
        <Button
          size="icon"
          variant="outline"
          className="rounded-full shadow-md"
          onClick={() => setOpen(true)}
          aria-label="Open feedback widget"
        >
          <MessageSquare className="h-4 w-4" />
        </Button>
      )}
    </div>
  );
}
```

```python
# In admin.py — add:
from datetime import timedelta

@router.get("/pulse")
async def get_pulse(
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> dict:
    """5 key numbers for the platform health view."""
    now = datetime.now(UTC)
    day_ago = now - timedelta(hours=24)
    week_ago = now - timedelta(days=7)

    # Active students (last 24h)
    active_result = await db.execute(
        select(func.count(func.distinct(AgentAction.student_id)))
        .where(AgentAction.created_at >= day_ago)
    )
    active_students = active_result.scalar() or 0

    # Agent calls (last 24h)
    calls_result = await db.execute(
        select(func.count(AgentAction.id))
        .where(AgentAction.created_at >= day_ago)
    )
    agent_calls = calls_result.scalar() or 0

    # Avg evaluation score (last 24h)
    score_result = await db.execute(
        select(func.avg(AgentAction.evaluation_score))
        .where(
            AgentAction.created_at >= day_ago,
            AgentAction.evaluation_score.isnot(None),
        )
    )
    avg_score = round(float(score_result.scalar() or 0), 2)

    # New enrollments (last 7 days)
    from app.models.enrollment import Enrollment
    enroll_result = await db.execute(
        select(func.count(Enrollment.id))
        .where(Enrollment.created_at >= week_ago)
    )
    new_enrollments = enroll_result.scalar() or 0

    # Open feedback items
    feedback_result = await db.execute(
        select(func.count(Feedback.id))
        .where(Feedback.resolved == False)  # noqa: E712
    )
    open_feedback = feedback_result.scalar() or 0

    log.info("admin.pulse_viewed", active_students=active_students)
    return {
        "active_students_24h": active_students,
        "agent_calls_24h": agent_calls,
        "avg_eval_score_24h": avg_score,
        "new_enrollments_7d": new_enrollments,
        "open_feedback": open_feedback,
    }
```

- [ ] **Step 3: Create admin/pulse/page.tsx**

Server component — 5 stat cards in a grid:

```tsx
// frontend/src/app/admin/pulse/page.tsx
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { cookies } from "next/headers";

interface PulseData {
  active_students_24h: number;
  agent_calls_24h: number;
  avg_eval_score_24h: number;
  new_enrollments_7d: number;
  open_feedback: number;
}

async function getPulse(): Promise<PulseData> {
  const cookieStore = await cookies();
  const token = cookieStore.get("access_token")?.value;
  const resp = await fetch(
    `${process.env.NEXT_PUBLIC_API_URL}/api/v1/admin/pulse`,
    {
      headers: { Authorization: `Bearer ${token}` },
      cache: "no-store",
    }
  );
  if (!resp.ok) throw new Error("Failed to load pulse data");
  return resp.json();
}

const STATS = (data: PulseData) => [
  { label: "Active students (24h)", value: data.active_students_24h },
  { label: "Agent calls (24h)", value: data.agent_calls_24h },
  { label: "Avg eval score (24h)", value: `${(data.avg_eval_score_24h * 100).toFixed(0)}%` },
  { label: "New enrollments (7d)", value: data.new_enrollments_7d },
  { label: "Open feedback", value: data.open_feedback },
];

export default async function PulsePage() {
  const data = await getPulse();
  return (
    <div className="p-6">
      <h1 className="mb-6 text-xl font-semibold">Platform Pulse</h1>
      <div className="grid grid-cols-2 gap-4 sm:grid-cols-3 lg:grid-cols-5">
        {STATS(data).map(({ label, value }) => (
          <Card key={label}>
            <CardHeader className="pb-1">
              <CardTitle className="text-xs font-medium text-muted-foreground">
                {label}
              </CardTitle>
            </CardHeader>
            <CardContent>
              <p className="text-2xl font-bold">{value}</p>
            </CardContent>
          </Card>
        ))}
      </div>
    </div>
  );
}
```
