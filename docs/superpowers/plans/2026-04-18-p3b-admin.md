```python
        select(AgentAction)
```

```tsx
// frontend/src/app/admin/audit-log/page.tsx
import { cookies } from "next/headers";

interface AuditItem {
  id: string;
  student_id: string;
  agent_name: string;
  task: string;
  evaluation_score: number | null;
  created_at: string;
}

async function getAuditLog(): Promise<AuditItem[]> {
  const cookieStore = await cookies();
  const token = cookieStore.get("access_token")?.value;
  const resp = await fetch(
    `${process.env.NEXT_PUBLIC_API_URL}/api/v1/admin/audit-log?limit=100`,
    {
      headers: { Authorization: `Bearer ${token}` },
      cache: "no-store",
    }
  );
  if (!resp.ok) return [];
  return resp.json();
}

export default async function AuditLogPage() {
  const items = await getAuditLog();
  return (
    <div className="p-6">
      <h1 className="mb-4 text-xl font-semibold">Audit Log</h1>
      {items.length === 0 ? (
        <p className="text-sm text-muted-foreground">No agent actions recorded yet.</p>
      ) : (
        <div className="overflow-x-auto">
          <table className="w-full text-sm">
            <thead>
              <tr className="border-b border-border text-left text-xs font-medium uppercase text-muted-foreground">
                <th className="pb-2 pr-4">Agent</th>
                <th className="pb-2 pr-4">Task</th>
                <th className="pb-2 pr-4">Score</th>
                <th className="pb-2">Time</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-border">
              {items.map((item) => (
                <tr key={item.id}>
                  <td className="py-2 pr-4 font-mono text-xs">{item.agent_name}</td>
                  <td className="max-w-xs truncate py-2 pr-4 text-xs">{item.task}</td>
                  <td className="py-2 pr-4 text-xs">
                    {item.evaluation_score != null
                      ? (item.evaluation_score * 100).toFixed(0) + "%"
                      : "—"}
                  </td>
                  <td className="py-2 text-xs text-muted-foreground">
                    {new Date(item.created_at).toLocaleString()}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
}
```

Count confusion-related agent actions per lesson. A "confusion event" is an `AgentAction` where `task` contains "confused", "don't understand", or where `agent_name = 'socratic_tutor'`:

```python
# In admin.py — add:
class LessonPerformance(PydanticModel):
    lesson_id: str
    lesson_title: str
    question_count: int
    confusion_count: int

@router.get("/content-performance", response_model=list[LessonPerformance])
async def get_content_performance(
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> list[LessonPerformance]:
    """Per-lesson confusion and question event counts."""
    from app.models.lesson import Lesson
    from sqlalchemy import case
    import structlog
    log = structlog.get_logger()

    result = await db.execute(
        select(
            AgentAction.metadata["lesson_id"].astext.label("lesson_id"),
            func.count(AgentAction.id).label("total"),
            func.sum(
                case(
                    (AgentAction.agent_name == "socratic_tutor", 1),
                    else_=0,
                )
            ).label("confusion_count"),
        )
        .where(AgentAction.metadata["lesson_id"].astext.isnot(None))
        .group_by("lesson_id")
        .order_by(func.count(AgentAction.id).desc())
        .limit(50)
    )
    rows = result.all()
    # Fetch lesson titles
    lesson_ids = [r.lesson_id for r in rows]
    lessons_result = await db.execute(
        select(Lesson.id, Lesson.title).where(
            Lesson.id.in_([uuid.UUID(lid) for lid in lesson_ids if lid])
        )
    )
    title_map = {str(r.id): r.title for r in lessons_result.all()}
    log.info("admin.content_performance_viewed", lesson_count=len(rows))
    return [
        LessonPerformance(
            lesson_id=r.lesson_id,
            lesson_title=title_map.get(r.lesson_id, r.lesson_id),
            question_count=int(r.total),
            confusion_count=int(r.confusion_count or 0),
        )
        for r in rows
    ]
```

```python
# In admin.py:
from app.models.course import Course
from app.models.exercise import Exercise

class CourseUpdateRequest(PydanticModel):
    title: str | None = None
    description: str | None = None
    metadata: dict | None = None

class ExerciseRubricUpdateRequest(PydanticModel):
    rubric: dict
    test_cases: list[dict] | None = None

@router.patch("/courses/{course_id}", status_code=204)
async def update_course(
    course_id: uuid.UUID,
    body: CourseUpdateRequest,
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> None:
    result = await db.execute(select(Course).where(Course.id == course_id))
    course = result.scalar_one_or_none()
    if not course:
        raise HTTPException(status_code=404, detail="Course not found")
    if body.title is not None:
        course.title = body.title
    if body.description is not None:
        course.description = body.description
    if body.metadata is not None:
        course.metadata = body.metadata
    await db.commit()
    import structlog
    structlog.get_logger().info("admin.course_updated", course_id=str(course_id))

@router.patch("/exercises/{exercise_id}/rubric", status_code=204)
async def update_exercise_rubric(
    exercise_id: uuid.UUID,
    body: ExerciseRubricUpdateRequest,
    db: AsyncSession = Depends(get_db),
    _: User = Depends(_require_admin),
) -> None:
    result = await db.execute(select(Exercise).where(Exercise.id == exercise_id))
    exercise = result.scalar_one_or_none()
    if not exercise:
        raise HTTPException(status_code=404, detail="Exercise not found")
    exercise.rubric = body.rubric
    if body.test_cases is not None:
        exercise.test_cases = body.test_cases
    await db.commit()
    structlog.get_logger().info("admin.rubric_updated", exercise_id=str(exercise_id))
```

```tsx
// frontend/src/app/admin/courses/[id]/edit/page.tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { useToast } from "@/lib/toast";

export default function CourseEditPage({ params }: { params: { id: string } }) {
  const [courseJson, setCourseJson] = useState("{}");
  const [rubricJson, setRubricJson] = useState("{}");
  const [saving, setSaving] = useState(false);
  const { toast } = useToast();

  const saveMetadata = async () => {
    try {
      JSON.parse(courseJson); // validate JSON before sending
    } catch {
      toast({ title: "Invalid JSON", variant: "destructive" });
      return;
    }
    setSaving(true);
    await fetch(`/api/v1/admin/courses/${params.id}`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: courseJson,
    });
    setSaving(false);
    toast({ title: "Course metadata saved" });
  };

  return (
    <div className="mx-auto max-w-2xl space-y-6 p-6">
      <h1 className="text-xl font-semibold">Edit Course</h1>
      <section className="space-y-2">
        <label className="text-sm font-medium" htmlFor="course-json">
          Course metadata (JSON)
        </label>
        <Textarea
          id="course-json"
          value={courseJson}
          onChange={(e) => setCourseJson(e.target.value)}
          rows={10}
          className="font-mono text-xs"
          aria-label="Course metadata JSON"
        />
        <Button onClick={saveMetadata} disabled={saving}>
          {saving ? "Saving…" : "Save metadata"}
        </Button>
      </section>
    </div>
  );
}
```
